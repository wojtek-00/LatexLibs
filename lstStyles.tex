% --------------------------------------------------
% GLOBAL LISTINGS SETTINGS (applied to all code)
% --------------------------------------------------
\lstset{
	language=[LaTeX]TeX,
	basicstyle=\ttfamily\small,
	backgroundcolor=\color{gray!10},
	showspaces=false,
	keepspaces=false,
	frame=single, 
	breaklines=true,
	captionpos=b,
	numbers=none,
	numberstyle=\tiny,
	keywordstyle=\color{blue},
	commentstyle=\color{gray},
	upquote=true
}

% ==================================================
%                  C++ LISTING STYLE
% ==================================================
%  Special highlighting for methods and types:
%   / ... |   → highlights methods in orange
%   * ... +   → highlights types in light blue
% ==================================================
\definecolor{LightBlue}{rgb}{0.678,0.847,0.902}
\definecolor{Orange}{rgb}{1.0,0.647,0.0}


\lstdefinestyle{cppMethod}{
	language=C++,
	basicstyle=\ttfamily,
	keywordstyle=\color{LightBlue}\bfseries,     % keywords like int, while, return
	identifierstyle=\color{black},               % all other identifiers
	emph={if,else,else if,while,true},
	emphstyle=\color{violet}\bfseries,
	moredelim=**[is][\color{Orange}]{/}{|},      % delimiters for methods
	moredelim=**[is][keywordstyle]{*}{+},        % delimiters for types
	literate={~}{{\ }}1
}

% Inline C++ method formatting command
% Usage: \cppMethod{*Type+./method|(args)}
\newcommand{\cppMethod}[1]{%
	\hspace*{-0.15em}\lstinline[style=cppMethod]!#1! % slight left shift for alignment
}

% ==================================================
%                  SQL LISTING STYLE
% ==================================================
\definecolor{SQLlightGreen}{rgb}{0.512, 0.775, 0.443}
\definecolor{SQLlightOrange}{rgb}{0.775, 0.502, 0.094}
\definecolor{SQLlightCyan}{rgb}{0.133, 0.592, 0.853}

\lstdefinelanguage{SQL}{
	keywords={
		CREATE, TABLE, PRIMARY, KEY, FOREIGN, REFERENCES, CONSTRAINT, ADD, OR, CHECK, FROM, SELECT, DISTINCT, AS, ORDER, BY, WHERE, LIKE, INSERT, INTO, VALUES, IS, NULL, AND
	},
	keywordstyle=\color{SQLlightGreen}\bfseries,
	morekeywords=[2]{INTEGER, VARCHAR, DATE, TIME},
	keywordstyle=[2]\color{SQLlightOrange}\bfseries,
	comment=[l]{--},
	morecomment=[s]{/*}{*/},
	commentstyle=\color{gray}\itshape,
	stringstyle=\color{SQLlightCyan},
	morestring=[b]',
	morestring=[b]"
}

\lstdefinestyle{sqlstyle}{
	language=SQL,
	basicstyle=\ttfamily\small,
	showstringspaces=false,
	columns=flexible,
	frame=single,
	breaklines=true,
	tabsize=2,
	captionpos=t,
	autogobble=true
}

% SQL environment shortcut
% Usage:
% \begin{sql}
%   SELECT * FROM STUDENT;
% \end{sql}
\lstnewenvironment{sql}{\lstset{style=sqlstyle}}{}
